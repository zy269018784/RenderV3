20enum class RHIAccess
21{
22 // Used when the previous state of a resource is not known,
23 // which implies we have to flush all GPU caches etc.
24 Unknown = 0,
25
26 // Read states
27 CPURead = 1 << 0,
28 Present = 1 << 1,
29 IndirectArgs = 1 << 2,
30 VertexOrIndexBuffer = 1 << 3,
31 SRVCompute = 1 << 4,
32 SRVGraphics = 1 << 5,
33 CopySrc = 1 << 6,
34 ResolveSrc = 1 << 7,
35 DSVRead = 1 << 8,
36
37 // Read-write states
38 UAVCompute = 1 << 9,
39 UAVGraphics = 1 << 10,
40 RTV = 1 << 11,
41 CopyDest = 1 << 12,
42 ResolveDst = 1 << 13,
43 DSVWrite = 1 << 14,
44
45 // Ray tracing acceleration structure states.
46 // Buffer that contains an AS must always be in either of these states.
47 // BVHRead -- required for AS inputs to build/update/copy/trace commands.
48 // BVHWrite -- required for AS outputs of build/update/copy commands.
49 BVHRead = 1 << 15,
50 BVHWrite = 1 << 16,
51
52 // Invalid released state (transient resources)
53 Discard = 1 << 17,
54
55 // Shading Rate Source
56 ShadingRateSource = 1 << 18,
57
58 Last = ShadingRateSource,
59 None = Unknown,
60 Mask = (Last << 1) - 1,
61
62 // A mask of the two possible SRV states
63 SRVMask = SRVCompute | SRVGraphics,
64
65 // A mask of the two possible UAV states
66 UAVMask = UAVCompute | UAVGraphics,
67
68 // A mask of all bits representing read-only states which cannot be combined with other write states.
69 ReadOnlyExclusiveMask = CPURead | Present | IndirectArgs | VertexOrIndexBuffer | SRVGraphics | SRVCompute | CopySrc | ResolveSrc | BVHRead,
70
71 // A mask of all bits representing read-only states which may be combined with other write states.
72 ReadOnlyMask = ReadOnlyExclusiveMask | DSVRead | ShadingRateSource,
73
74 // A mask of all bits representing readable states which may also include writable states.
75 ReadableMask = ReadOnlyMask | UAVMask,
76
77 // A mask of all bits representing write-only states which cannot be combined with other read states.
78 WriteOnlyExclusiveMask = RTV | CopyDest | ResolveDst,
79
80 // A mask of all bits representing write-only states which may be combined with other read states.
81 WriteOnlyMask = WriteOnlyExclusiveMask | DSVWrite,
82
83 // A mask of all bits representing writable states which may also include readable states.
84 WritableMask = WriteOnlyMask | UAVMask | BVHWrite
85};
