		set (GLFW_INCLUDE_PATH	${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${GLFW_DIRECTORY}/include)
	else ()                                                                                 
		set (GLFW_LIBS)                                                                      
		set (GLFW_LINK_PATH		${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${GLFW_DIRECTORY}/lib)
		set (GLFW_INCLUDE_PATH	${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${GLFW_DIRECTORY}/include)
	
	endif ()
endif ()
