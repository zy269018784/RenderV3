option (USE_Vulkan "USE Vulkan" ON)
if (USE_Vulkan)
	set (Vulkan_DIRECTORY "Vulkan")
	if (CMAKE_SYSTEM_NAME STREQUAL Windows)
		set (Vulkan_INCLUDE_PATH  ${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${Vulkan_DIRECTORY}/include )
		set (Vulkan_LINK_PATH     ${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${Vulkan_DIRECTORY}/lib)
		set (Vulkan_LIBS vulkan-1.lib)
	elseif (CMAKE_SYSTEM_NAME STREQUAL Linux)
		set (Vulkan_LIBS libvulkan.so)
	endif ()		
endif ()
